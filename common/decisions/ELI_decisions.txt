

political_actions = {
	ELI_elect_a_leader = {

		icon = GFX_decision_generic_authority

		available = {
			OR = {
				is_subject = no
				threat > 0.25
				has_war = yes
				AND = {
					ELU = { NOT = { has_country_leader = ELU_rerirhalls } }
					threat > 0.05
				}
			}
		}

		allowed = {
			original_tag = ELI
		}

		visible = {
			has_country_leader = ELI_mountainhalls
		}

		fire_only_once = yes

		cost = 50

		complete_effect = {
			retire_character = ELI_mountainhalls
		}

		ai_will_do = {
			base = 1000
		}
	}

	ELI_unite_the_realm = {

		icon = GFX_decision_dwarves_elves_men

		available = {
			OR = {
				is_subject = no
				threat > 0.25
				has_war = yes
				AND = {
					ELU = { NOT = { has_idea = ELU_disjointed_purpose } }
					threat > 0.15
				}
				AND = {
					ELU = { NOT = { has_idea = ELU_disjointed_purpose } }
					has_stability > 0.5
				}
			}
		}

		allowed = {
			original_tag = ELI
		}

		visible = {
			has_idea = ELI_disjointed_purpose
		}

		fire_only_once = yes

		cost = 50

		complete_effect = {
			remove_ideas = ELI_disjointed_purpose
		}

		ai_will_do = {
			base = 1000
		}
	}
