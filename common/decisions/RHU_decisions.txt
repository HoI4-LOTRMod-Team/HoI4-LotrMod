RHU_kill_the_blue_wizards_cat = {
	RHU_kill_alatar = {
		icon = GFX_decision_generic_civil_support
		cost = 50

        available = {
			OR = {
				has_idea = RHU_mission_of_the_blue_wizards
				has_idea = RHU_mission_of_the_blue_wizards_1
			}
        }

		visible = {
			has_civil_war = no
			NOT = {
				has_global_flag = rhun_alatar_has_fallen
			}
		}

		days_remove = 20
        fixed_random_seed = no

		remove_effect = {
			random_list = {
				40 = {
					modifier = {
						factor = 0.8
						is_ai = yes
					}
					add_stability = -0.05
					add_popularity = {
						ideology = democratic
						popularity = 0.05
					}
				}
				60 = {
					modifier = {
						factor = 1.2
						is_ai = yes
					}
					add_stability = -0.05
					add_popularity = {
						ideology = fascism
						popularity = 0.05
					}
					set_global_flag = rhun_alatar_has_fallen
					if = {
						limit = {
							has_idea = RHU_mission_of_the_blue_wizards
						}
						swap_ideas = { 
							remove_idea = RHU_mission_of_the_blue_wizards 
							add_idea = RHU_mission_of_the_blue_wizards_1
						}
					} else = {
						remove_ideas = RHU_mission_of_the_blue_wizards_1
					}
				}
			}
		}
		ai_will_do = {
			factor = 7
		}
	}
