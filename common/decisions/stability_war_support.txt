demobilization = {
	demobilization_economic = {

		icon = GFX_decision_revolt

		is_good = no

		available = {
			has_unsupported_centralization_law = yes
			has_war = no
		}

		activation = {
			has_unsupported_centralization_law = yes
			has_war = no
		}
		cost = 50

		visible = {
			has_war = no
			has_unsupported_centralization_law = yes
		}
		complete_effect = {
			if = {
				limit = {
					has_war_support < 0.15
				}
				add_ideas = weak_centralization
			}
			else_if = {
				limit = {
					has_war_support < 0.25
				}
				add_ideas = moderate_centralization
			}
			else_if = {
				limit = {
					has_war_support < 0.5
				}
				add_ideas = strong_centralization
			}
			else_if = {
				limit = {
					has_war_support < 0.8
				}
				add_ideas = very_strong_centralization
			}
			if = {
				limit = {
					has_idea = strikes_1
				}
				remove_ideas = strikes_1
			}
			if = {
				limit = {
					has_idea = strikes_2
				}
				remove_ideas = strikes_2
			}
			if = {
				limit = {
					has_idea = strikes_3
				}
				remove_ideas = strikes_3
			}
		}
		ai_will_do = {
			factor = 500 #should really get on this
		}

	}
	demob_economic_mission = {
		available = {
			OR = {
				has_unsupported_centralization_law = no
				has_war = yes
			}
		}

		activation = {
			has_unsupported_centralization_law = yes
			has_war = no
		}
		icon = revolt
		is_good = no
		days_mission_timeout = 120
		complete_effect = {
			if = {
				limit = {
					has_idea = strikes_1
				}
				remove_ideas = strikes_1
			}
			if = {
				limit = {
					has_idea = strikes_2
				}
				remove_ideas = strikes_2
			}
			if = {
				limit = {
					has_idea = strikes_3
				}
				remove_ideas = strikes_3
			}
		}
		timeout_effect = {
			activate_mission = demob_economic_mission
			if = {
				limit = {
					has_idea = strikes_3
					has_stability < 0.21
					NOT = {
						has_government = revolutionary
					}
				}
				start_civil_war = {
					ideology = revolutionary
				}
			}
			if = {
				limit = {
					has_idea = strikes_3
					has_stability < 0.21
					has_government = revolutionary

				}
				start_civil_war = {
					ideology = cooperative
				}
			}
			if = {
				limit = {
					has_idea = strikes_3
					has_stability < 0.41
					NOT = {
						has_government = revolutionary
					}
				}
				add_popularity = { ideology = revolutionary popularity = 0.1}
			}
			if = {
				limit = {
					has_idea = strikes_3
					has_stability < 0.41
					has_government = revolutionary

				}
				add_popularity = { ideology = cooperative popularity = 0.1}
			}
			if = {
				limit = {
					has_idea = strikes_3
					has_stability > 0.4
				}
				add_stability = -0.1
			}
			if = {
				limit = {
					has_idea = strikes_2
				}
				swap_ideas = {
					remove_idea = strikes_2
					add_idea = strikes_3
				}
			}
			if = {
				limit = {
					has_idea = strikes_1
				}
				swap_ideas = {
					remove_idea = strikes_1
					add_idea = strikes_2
				}
			}
			if = {
				limit = {
					NOT = {
						has_idea = strikes_1
						has_idea = strikes_2
						has_idea = strikes_3
					}
				}
				add_ideas = strikes_1
			}
		}
	}
	demobilization_manpower = {

		icon = GFX_decision_revolt

		icon = revolt
		is_good = no

		available = {
			has_unsupported_manpower_law = yes
			has_war = no
		}

		allowed = {
			always = yes
		}

		cost = 50
		ai_will_do = {
			factor = 500
		}


		visible = {
			has_unsupported_manpower_law = yes
			has_war = no

		}
		complete_effect = {
			if = {
				limit = {
					has_war_support < 0.1
				}
				add_ideas = volunteer_service
			}
			else_if = {
				limit = {
					has_war_support < 0.2
				}
				add_ideas = limited_levy
			}
			else_if = {
				limit = {
					has_war_support < 0.6
				}
				add_ideas = extensive_levy
			}
			else_if = {
				limit = {
					has_idea = slave_soldiers
					NOT = { has_government = belligerent }
				}
				add_ideas = elite_mercenaries
			}
			else = {
				add_ideas = mass_levy
			}
		}



	}

		icon = revolt
		is_good = no
		days_mission_timeout = 120
		available = {
			OR = {
				has_unsupported_manpower_law = no
				has_war = yes
			}
		}
		activation = {
			has_unsupported_manpower_law = yes
			has_war = no
		}
		visible = {
		}
		cancel_if_not_visible = yes
		timeout_effect = {
			if = {
				limit = {
					has_government = revolutionary
					has_stability < 0.2
				}
				start_civil_war = {
					ideology = cooperative
				}
			}
			if = {
				limit = {
					NOT = { has_government = revolutionary }
					has_stability < 0.3
				}
				start_civil_war = {
					ideology = revolutionary
				}
			}
			if = {
				limit = {
					has_stability > 0.2
				}
				add_stability = -0.1
			}
			if = {
				limit = {
					has_government = revolutionary
				}
				add_popularity = { ideology = cooperative popularity = 0.1 }
			}
			if = {
				limit = {
					NOT = { has_government = revolutionary }
				}
				add_popularity = { ideology = revolutionary popularity = 0.1 }
			}
		}
	}
