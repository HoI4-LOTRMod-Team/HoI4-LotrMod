

# Initialize the variables at game start
lthi_initialize_values = {
	set_variable = { lthi_nandor_influence = 100 }
	set_variable = { lthi_noldor_influence = 25 }
	set_variable = { lthi_sindar_influence = 20 }
	set_variable = { lthi_halfelves_influence = 5 }

	lthi_update_values = yes
}


# Call this every time one of the influence vars is changed
lthi_update_values = {
	set_temp_variable = { lthi_influence_sum = lthi_nandor_influence }
	add_to_temp_variable = { lthi_influence_sum = lthi_noldor_influence }
	add_to_temp_variable = { lthi_influence_sum = lthi_sindar_influence }
	add_to_temp_variable = { lthi_influence_sum = lthi_halfelves_influence }

	set_variable = { lthi_nandor_influence_fraction = lthi_nandor_influence }
	set_variable = { lthi_noldor_influence_fraction = lthi_noldor_influence }
	set_variable = { lthi_sindar_influence_fraction = lthi_sindar_influence }
	set_variable = { lthi_halfelves_influence_fraction = lthi_halfelves_influence }

	divide_variable = { lthi_nandor_influence_fraction = lthi_influence_sum }
	divide_variable = { lthi_noldor_influence_fraction = lthi_influence_sum }
	divide_variable = { lthi_sindar_influence_fraction = lthi_influence_sum }
	divide_variable = { lthi_halfelves_influence_fraction = lthi_influence_sum }
}
