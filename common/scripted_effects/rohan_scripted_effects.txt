improve_royal_guard = {
	
	custom_effect_tooltip = improve_royal_guard_tt

	hidden_effect = {
		if = {
			limit = { NOT = { has_country_flag = guard_improved } }

			set_country_flag = guard_improved

			add_equipment_to_stockpile = {
				type = heavy_cavalry_chassis_0
				amount = 150
			}

			add_equipment_to_stockpile = {
				type = motorized_equipment
				amount = 50
			}

			add_units_to_division_template = {
				template_name = "Royal Guard"
				regiments = {
					heavy_cavalry = 0
					heavy_cavalry = 0
					heavy_cavalry = 1
					heavy_cavalry = 1
					motorized = 2
					motorized = 2
				}
				support = {
					engineer = 0
				}
			}
		}

		else = {
			add_equipment_to_stockpile = {
				type = heavy_cavalry_chassis_0
				amount = 150
			}

			add_equipment_to_stockpile = {
				type = motorized_equipment
				amount = 50
			}

			add_units_to_division_template = {
				template_name = "Royal Guard"
				regiments = {
					heavy_cavalry = 3
					heavy_cavalry = 3
					heavy_cavalry = 3
					heavy_cavalry = 3
					motorized = 4
					motorized = 4
				}
			}
		}
	}
}
