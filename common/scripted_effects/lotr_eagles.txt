





lotr_set_eagles_cap = {
	set_variable = { lotr_max_eagle_count = 30 }
}


# @docs(Eagles): increases the maximum amount of eagles/fellbeasts that can be fielded by 15
increase_eagles_cap_15 = {
	add_to_variable = {
		var = lotr_max_eagle_count
		value = 15
		tooltip = increase_eagle_cap_tt
	}
}

	}
