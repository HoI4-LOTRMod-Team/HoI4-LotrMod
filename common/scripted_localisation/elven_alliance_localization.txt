defined_text = {
	name = DisplayElvenAllianceMembers
	text = {
		trigger = {
			if = {
				limit = { has_variable = temp_event_index }
				add_to_temp_variable = { temp_event_index = 1 }
			}
			else = {
				set_temp_variable = { temp_event_index = 0 }
			}

			check_variable = { temp_event_index < global.elven_alliance_members^num }
		}
		localization_key = "elven_alliance_memberlist_recruse"
	}
}



defined_text = {
	name = DisplayElvenAllianceMember
	text = {
		trigger = {
			check_variable = { global.elven_alliance_members^num > temp_event_index }
		}
		localization_key = "[?global.elven_alliance_members^temp_event_index.GetNameWithFlag]"
	}
}
