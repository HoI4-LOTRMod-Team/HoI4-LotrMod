	possible = {
		achievement_standard_possible_check = yes

		}
	}

			all_core_state = {
				OR = {
					is_controlled_by = GUN

				}
			}
		}
		MIR = {
			all_core_state = {
				OR = {
					is_controlled_by = GUN
					is_controlled_by = DGU
				}
			}
		}
		DAL = {
			all_core_state = {
				OR = {
					is_controlled_by = GUN

				}
			}
		}
	}
}
