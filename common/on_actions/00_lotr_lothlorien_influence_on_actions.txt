




# To change the bonuses of the factions correctly, we run this on a weekly basis

on_actions = {
	
	on_weekly_LTH  = {

		effect = {

			lthi_update_values = yes

		}

	}


	on_yearly_LTH = {
		effect = {
			if = {
				limit = { has_dynamic_modifier = { modifier = lth_preservationofknowledge_modifier } }
				add_to_variable = { lth_preservationofknowledge_bonus = 1 }
			}
		}
	}
}
