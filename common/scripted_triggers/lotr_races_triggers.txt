is_elves = {
	custom_trigger_tooltip = {
		tooltip = "is Elves"
		OR = {
			original_tag = LIN
			original_tag = LTH
			original_tag = RIV
			original_tag = HLI
			original_tag = NDH
			AND = {
				original_tag = MIR
				NOT = { has_country_flag = has_switched_to_spiders }
			}
			AND = {
				original_tag = LTH
				NOT = { has_country_flag = has_switched_to_orcs }
			}
		}
	}
}

# is the country is elven at the start?
is_elves_init = {
	OR = {
		original_tag = LIN
		original_tag = LTH
		original_tag = MIR
		original_tag = RIV
		original_tag = HLI
		original_tag = NDH
	}
}

is_men = {
	custom_trigger_tooltip = {
		tooltip = "is Men"
		NOT = {
			is_elves = yes
			is_orcs = yes
			is_hobbits = yes
			is_dwarves = yes
			is_spiders = yes
			is_ents = yes
			is_other = yes
		}
	}
}

is_men_init = {
	NOT = {
		is_elves_init = yes
		is_orcs_init = yes
		is_hobbits_init = yes
		is_dwarves_init = yes
		is_spiders_init = yes
		is_ents_init = yes
		is_other_init = yes
	}
}
