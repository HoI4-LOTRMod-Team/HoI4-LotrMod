# Suggest misty mountain pact
country_event = {
	id = misty_pact.3
	title = misty_pact.3.t
	desc = misty_pact.3.d
	picture = GFX_report_event_ring

	is_triggered_only = yes

	option = { #yes
		name = misty_pact.3.a
		ai_chance = {
			base = 25
			modifier = {
				factor = -5
				is_puppet = yes
			}
			modifier = {
				add = 5
				has_opinion = {
					target = FROM
					value > 50
				}
			}
			modifier = {
				add = 10
				has_non_aggression_pact_with = FROM
			}
			modifier = {
				factor = -999
				has_war_with = FROM
			}
		}
		add_opinion_modifier = { target = FROM modifier = small_increase }
		add_political_power = 50
		add_popularity = {
			ideology = cooperative
			popularity = 0.025
		}
		FROM = {
			country_event = { id = misty_pact.1 }
		}
		set_country_flag = MISTY_PACT
		hidden_effect = {
			ISE = {
				add_to_variable = { misty_pact_members = 1 }
			}
		}
	}

	option = { #no
		name = misty_pact.3.b
		ai_chance = {
			base = 5
			modifier = {
				factor = 999
				has_war_with = FROM
			}
			modifier = {
				add = 5
				has_opinion = {
					target = FROM
					value < 0
				}
			}
		}
		add_opinion_modifier = { target = FROM modifier = small_decrease }
		FROM = {
			country_event = { id = misty_pact.2 }
		}
	}
}
