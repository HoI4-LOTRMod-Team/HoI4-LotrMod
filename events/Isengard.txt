# Saruman Demands Enedwaith Total Submission
country_event = {
	id = isengard.4
	title = isengard.4.t
	desc = isengard.4.d
	picture = GFX_report_event_paper

	is_triggered_only = yes

	option = { # Accept
		name = isengard.4.a
		ai_chance = {
			base = 30
			modifier = {
				add = -25
				ISE = { has_army_size = { size < 30 } }
			}
			modifier = {
				add = 30
				ISE = { has_army_size = { size > 40 } }
			}
			modifier = {
				add = 100
				ENE = { has_government = fascism }
			}
		}
 		ISE = { annex_country = { target = ENE } }
	}
	option = { # Refuse
		name = isengard.4.b
		ai_chance = {
			base = 50
			modifier = {
				any_other_country = {
					has_guaranteed = ENE
				}
				factor = 5
			}
			modifier = {
				add = 20
				ENE = { has_army_size = { size > 20 } }
			}
			modifier = {
				factor = 0
				is_historical_focus_on = yes
			}
		}
		ENE = { declare_war_on = { target = ISE type = annex_everything } }
	}
}
