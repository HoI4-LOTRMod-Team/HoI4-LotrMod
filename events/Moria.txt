add_namespace = moria

# Mordor Civil War
country_event = {
	id = moria.1
	title = moria.1.t
	desc = moria.1.d
	picture = GFX_report_event_orcs_marching

	fire_only_once = yes

	trigger = {
		tag = MOA
		date > 3018.1.15
	}

	mean_time_to_happen = {
		days = 80
	}

	option = { # Support Dwarves
		name = moria.1.a
		ai_chance = {base = 10}

		kill_ideology_leader = neutrality

		set_global_flag = moria_civil_war
		ERE = { country_event = { days = 2 id = moria.2 } }
		add_ideas = MOA_civil_war
		hidden_effect = {
			set_capital = { state = 134 }
			start_civil_war = {
				ruling_party = democratic
				ideology = communism
				size = -1.0
				capital = 73
				states = { 73 }
			}
			random_other_country = {
				limit = {
					original_tag = MOA
					has_government = communism
				}
				load_oob = "MOA_Orcs" #in case we want special oob for communist mordor
				load_focus_tree = moria
				complete_national_focus = thefateofmoria
				add_ideas = service_by_requirement
				add_ideas = MOA_civil_war
				add_ideas = MOA_loose_balrog
				set_cosmetic_tag = MOA
			}
			news_event = { hours = 12 id = news.344 }

			load_oob = "MOA_Dwarves"

			add_popularity = {
				ideology = democratic
				popularity = 0.4
			}
		}
	}

	option = { # Support Orcs
		name = moria.1.b
		ai_chance = {base = 10}

		kill_ideology_leader = neutrality

		set_global_flag = moria_civil_war
		ERE = { country_event = { days = 2 id = moria.2 } }
		add_ideas = MOA_civil_war
		hidden_effect = {
			set_capital = { state = 134 }
			start_civil_war = {
				ruling_party = democratic
				ideology = communism
				size = -1.0
				capital = 73
				states = { 73 }
			}
			random_other_country = {
				limit = {
					original_tag = MOA
					has_government = communism
				}
				load_oob = "MOA_Orcs" #in case we want special oob for communist mordor
				load_focus_tree = moria
				complete_national_focus = thefateofmoria
				add_ideas = service_by_requirement
				add_ideas = MOA_civil_war
				add_ideas = MOA_loose_balrog
				set_cosmetic_tag = MOA
			}
			news_event = { hours = 12 id = news.344 }

			load_oob = "MOA_Dwarves"

			add_popularity = {
				ideology = democratic
				popularity = 0.4
			}
			#Exact same as option above but switch tag to the communists
			random_other_country = {
				limit = {
					original_tag = MOA
					has_government = communism
				}
				change_tag_from = MOA
			}
		}
	}
}
