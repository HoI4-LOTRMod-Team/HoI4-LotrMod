# Vote on the fellowship's path
country_event = {
	id = rivendell.19
	title = rivendell.19.t
	desc = rivendell.19.desc
	picture = GFX_report_event_elrond_council

	is_triggered_only = yes

	# through the redhorn pass
	option = {
		name = rivendell.19.a
		ai_chance = {
			base = 2
			# Galadriel magic
			modifier = {
				factor = 2
				has_global_flag = mirror_predicted_redhorn
			}
		}

		# human votes count for more, to give players mor agency in the case of a tie
		if = {
			limit = { is_ai = yes }
			add_to_variable = {
	    		var = global.fs_council_vote_redhorn
	    		value = 1
			}
		}
		else = {
			add_to_variable = {
	    		var = global.fs_council_vote_redhorn
	    		value = 1.5
			}
		}

		if = {
			limit = { tag = RIV }
			set_global_flag = fs_rivendell_responded
		}
		else_if = {
			limit = { tag = ERE }
			set_global_flag = fs_erebor_responded
		}
		else_if = {
			limit = { tag = GON }
			set_global_flag = fs_gondor_responded
		}
		else_if = {
			limit = { tag = MIR }
			set_global_flag = fs_mirkwood_responded
		}
		else_if = {
			limit = { tag = LTH }
			set_global_flag = fs_lothlorien_responded
		}

		ring_check_if_council_of_elrond_concluded = yes
	}
	# past isengard
	option = {
		name = rivendell.19.b
		ai_chance = {
			base = 1
			modifier = {
	    		factor = 3
	    		has_global_flag = gandalf_saruman_united
			}
			modifier = {
	    		factor = 0
	    		is_historical_focus_on = yes
			}
			modifier = {
				factor = 2
				has_global_flag = mirror_predicted_isengard
			}
		}

		if = {
			limit = { is_ai = yes }
			add_to_variable = {
	    		var = global.fs_council_vote_isengard
	    		value = 1
			}
		}
		else = {
			add_to_variable = {
	    		var = global.fs_council_vote_isengard
	    		value = 1.5
			}
		}

		if = {
			limit = { tag = RIV }
			set_global_flag = fs_rivendell_responded
		}
		else_if = {
			limit = { tag = ERE }
			set_global_flag = fs_erebor_responded
		}
		else_if = {
			limit = { tag = GON }
			set_global_flag = fs_gondor_responded
		}
		else_if = {
			limit = { tag = MIR }
			set_global_flag = fs_mirkwood_responded
		}
		else_if = {
			limit = { tag = LTH }
			set_global_flag = fs_lothlorien_responded
		}

		ring_check_if_council_of_elrond_concluded = yes
	}
	# past gundabad
	option = {
		name = rivendell.19.c
		ai_chance = {
			base = 1
			modifier = {
        		factor = 2
        		ANG = { has_completed_focus = ANG_join_gondor }
    		}
    		modifier = {
        		factor = 2
        		ANG = { is_ai = no }
    		}
    		modifier = {
        		factor = 0
        		is_historical_focus_on = yes
    		}
			modifier = {
				factor = 2
				has_global_flag = mirror_predicted_gundabad
			}
		}

		if = {
			limit = { is_ai = yes }
			add_to_variable = {
	    		var = global.fs_council_vote_angmar
	    		value = 1
			}
		}
		else = {
			add_to_variable = {
	    		var = global.fs_council_vote_angmar
	    		value = 1.5
			}
		}

		if = {
			limit = { tag = RIV }
			set_global_flag = fs_rivendell_responded
		}
		else_if = {
			limit = { tag = ERE }
			set_global_flag = fs_erebor_responded
		}
		else_if = {
			limit = { tag = GON }
			set_global_flag = fs_gondor_responded
		}
		else_if = {
			limit = { tag = MIR }
			set_global_flag = fs_mirkwood_responded
		}
		else_if = {
			limit = { tag = LTH }
			set_global_flag = fs_lothlorien_responded
		}

		ring_check_if_council_of_elrond_concluded = yes
	}
	# neutral vote / dont care
	option = {
		name = rivendell.19.d
		ai_chance = {
			base = 0
		}

		if = {
			limit = { tag = RIV }
			set_global_flag = fs_rivendell_responded
		}
		else_if = {
			limit = { tag = ERE }
