# Proposal of construction of embassy

	id = dorwinion.2
	title = dorwinion.2.t
	desc = dorwinion.2.d
	picture = GFX_report_event_ring

	is_triggered_only = yes

	option = { #yes
		name = dorwinion.2.a
		ai_chance = {
			base = 40
			modifier = {
				add = 20
				has_opinion = {
					target = RHU
					value > 25
				}
			}
			modifier = {
				add = 20
				fascism > 0.20
			}
		}
		DOR = {
			add_popularity = {

		RHU = {
			add_opinion_modifier = { target = DOR modifier = medium_increase }

		}

	option = { #no
