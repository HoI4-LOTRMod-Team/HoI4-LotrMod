# Class Conflicts Divide [Root.GetName]
country_event = {
	id = political.4
	title = political.4.t
	desc = political.4.d
	picture = GFX_report_event_gathering_protest

	trigger = {
		OR = { tag = VAL tag = DAL tag = HAR tag = DAG tag = RHU tag = CAR tag = RHD tag = LIN tag = ENT tag = ENE tag = SPI tag = UMB tag = PEL tag = KND tag = ITH }
		has_idea_with_trait = communist_revolutionary
		NOT = { has_government = revolutionary }
		revolutionary > 0.2
	}

	mean_time_to_happen = {
		days = 365
		modifier = {
			factor = 0.5
			has_country_flag = revolutionary_popular_movement
		}
	}

	option = {
		name = political.4.a
		add_popularity = {
			ideology = revolutionary
			popularity = 0.05
		}
		add_stability = -0.05
	}
}
