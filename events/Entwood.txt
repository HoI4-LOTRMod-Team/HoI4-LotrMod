		date > 3020.3.1

	}

		days = 200	
	}

		name = entwood.1.a
		ai_chance = {
			base = 15
			modifier = {
				add = 30
				ISE = { has_war_with = ROH }
			}
			modifier = {
				add = 15
				ISE = { has_government = fascism }
			}
			modifier = {
				add = 50
				has_global_flag = ents_encouraged
			}
			modifier = {
				add = 50

			}
		}
		declare_war_on = { target = ISE type = puppet_wargoal_focus } 
		hidden_effect = { news_event = { hours = 10 id = news.349 }}
	}
	option = { #Refuse
		name = entwood.1.b
		ai_chance = {
			base = 75
		}
	}
	
}
